Delete an account by username or UUID.

**** Delete by username

#+begin_src sh
export ORES_CLI_DB_PASSWORD
./ores.cli iam accounts delete ${db_args} ${log_args} \
  --key newuser4
#+end_src

#+RESULTS:

#+end_src
