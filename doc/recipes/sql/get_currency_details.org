#+begin_src sql
select iso_code, name, symbol, fraction_symbol,
       fractions_per_unit, rounding_type, rounding_precision, format
from ores.currencies
where iso_code = 'USD'
      and valid_to = '9999-12-31 23:59:59'::timestamptz;
#+end_src
