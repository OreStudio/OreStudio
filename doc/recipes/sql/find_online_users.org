#+title: Find Online Users
#+description: Find Online Users
#+type: recipe
#+level: cross
#+filetags: :recipe:sql:login_info:
#+created: 2026-05-21
#+updated: 2026-05-21

#+begin_src sql
select a.username, li.last_login, li.last_ip
from ores.login_info li
join ores.accounts a on a.id = li.account_id
  and a.valid_to = '9999-12-31 23:59:59'::timestamptz
where li.online = 1;
#+end_src
