** List all as JSON

#+begin_src sh
export ORES_CLI_DB_PASSWORD
./ores.cli dq change-reasons list ${db_args} ${log_args} --format json | jq .[:5]
#+end_src

** List specific change reason

#+begin_src sh
export ORES_CLI_DB_PASSWORD
./ores.cli dq change-reasons list ${db_args} ${log_args} \
  --format json --key "data_correction" | jq .
#+end_src
