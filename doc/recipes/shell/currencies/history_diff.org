#+title: History Diff
#+description: Show currency version history as a unified diff via currencies history-diff.
#+type: recipe
#+level: cross
#+filetags: :recipe:shell:currencies:
#+created: 2026-06-05
#+updated: 2026-06-05

Show currency version history as a unified diff. The +/- rows come
straight from the server-computed changes in the history response —
the shell performs no comparison of its own.

#+name: currencies-history-diff
#+begin_src ores-shell :exports none
currencies history-diff USD
exit
#+end_src

#+begin_src sh :noweb yes :results output  :wrap example
export ORES_SHELL_LOGIN_PASSWORD
./ores.shell ${log_args} ${connect_args} ${login_args}  << 'EOF'
<<currencies-history-diff>>
EOF
#+end_src
