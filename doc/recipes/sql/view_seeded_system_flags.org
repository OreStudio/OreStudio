List all system flags seeded by =system_flags_populate.sql=:

#+begin_src sql
select name, enabled, description
from ores.feature_flags
where name like 'system.%'
  and valid_to = '9999-12-31 23:59:59'::timestamptz
order by name;
#+end_src
