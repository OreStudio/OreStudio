  --alpha2-code "XX" \
  --alpha3-code "XXX" \
  --name "Test Country" \
  --numeric-code "999" \
  --official-name "The Republic of Test Country" \
  --recorded-by admin \
  --change-reason-code "system.new_record" \
  --change-commentary "Adding test country via CLI"
